		function play(){
			var audio = new Audio("sonidos/sonido_1.mp3");
			audio.play();
		};	
        var names_of_weapon_list = ["MINI GUN", "SNIPER", "SHOTGUN"]; // class nombre_arma
		var weapons = [
			'<img src="imagenes/arma_1.png" width="200" height="100"/>', 
			'<img src="imagenes/arma_2.png" width="200" height="100"/>',
			'<img src="imagenes/arma_3.png" width="200" height="100"/>']; // class arma
            var posicion = 0;
        function arma_derecha(){
            if(posicion == 2){
                posicion = 0;
            }
            else{
                posicion += 1;
            };
            escribir();
			cambio_stats();
			clear();
        };
        function arma_izquierda(){
            if(posicion == 0){
                posicion = 2;
            }
            else{
                posicion -= 1;
            };
            escribir();
			cambio_stats();
			clear();
        };			
        function escribir(){
            document.getElementsByClassName("nombrearma")[0].innerHTML=names_of_weapon_list[posicion];
			document.getElementsByClassName("arma")[0].innerHTML=weapons[posicion];
        };
        escribir();
		var ctx_range = document.getElementsByClassName("rangebarracanvas")[0].getContext("2d");
		var ctx_rpm = document.getElementsByClassName("rpmbarracanvas")[0].getContext("2d");
		var ctx_damage = document.getElementsByClassName("damagebarracanvas")[0].getContext("2d");
		var ctx_charger_size = document.getElementsByClassName("chargerbarracanvas")[0].getContext("2d");
		var width_range = 0;
		var width_rpm = 0;
		var width_damage = 0;
		var width_charger = 0;
		var cuanto_range = 0;
		var cuanto_rpm = 0;
		var cuanto_damage = 0;
		var cuanto_charger = 0;
		function cambio_stats(){
			if(posicion == 0){
				cuanto_range = 100;
				cuanto_rpm = 210;
				cuanto_damage = 72;
				cuanto_charger = 150;
			};
			if(posicion == 1){
				cuanto_range = 320;
				cuanto_rpm = 50;
				cuanto_damage = 270;
				cuanto_charger = 20;
			};
			if(posicion == 2){
				cuanto_range = 150;
				cuanto_rpm = 160;
				cuanto_damage = 100;
				cuanto_charger = 70;
			};				
		};
		function draw(){
			ctx_range.fillStyle = "#CC3";
			ctx_range.fillRect(0, 0, width_range, 10);
			ctx_rpm.fillStyle = "#CC3";
			ctx_rpm.fillRect(0, 0, width_rpm, 10);
			ctx_damage.fillStyle = "#CC3";
			ctx_damage.fillRect(0, 0, width_damage, 10);
			ctx_charger_size.fillStyle = "#CC3";
			ctx_charger_size.fillRect(0, 0, width_charger, 10);
			if(width_range <= cuanto_range){
				width_range += 10;
			};
			if(width_charger <= cuanto_charger){
				width_charger += 10;
			};
			if(width_rpm <= cuanto_rpm){
				width_rpm += 10;
			};
			if(width_damage <= cuanto_damage){
				width_damage += 10;
			};
		};
		var es_chrome = navigator.userAgent.toLowerCase().indexOf('chrome') > -1;
		var es_firefox = navigator.userAgent.toLowerCase().indexOf('firefox') > -1;
		var es_opera = navigator.userAgent.toLowerCase().indexOf('opera');
		function checar_rendimiento(){
			if(es_firefox){
				speed = 15;
				return speed;
			};
			if(es_opera){
				speed = 15;
				return speed;
			};
			if(es_chrome){
				speed = 15;
				return speed;
			};
			if (document.documentMode || /Edge/.test(navigator.userAgent)) {
				speed = .1;
				return speed;
			};
		};
		setInterval(draw, checar_rendimiento());
		function clear(){
			width_range = 0;
			width_rpm = 0;
			width_damage = 0;
			width_charger = 0;
			ctx_range.clearRect(0, 0, 340, 10);
			ctx_rpm.clearRect(0, 0, 340, 10);
			ctx_damage.clearRect(0, 0, 340, 10);
			ctx_charger_size.clearRect(0, 0, 340, 10);
			draw();
		};
		cambio_stats();
		draw();
    </script>
